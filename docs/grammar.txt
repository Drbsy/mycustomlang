# === PROGRAM ENTRY ===
program                ::= global_statement_list [ TOK_EOF ]

# === STATEMENTS ===
global_statement_list  ::= global_statement { global_statement }
global_statement       ::= function_def | var_def

local_statement_list   ::= local_statement { local_statement }
local_statement        ::= var_def | return_stmt

op_local_statement_list ::= local_statement_list | ε

# === VARIABLE DECLARATION ===
var_def                ::= TOK_VAR TOK_ID variable_type TOK_ASSIGN expr_value
variable_type          ::= explicit_type | implicit_type
explicit_type          ::= TOK_COLON type_definition
implicit_type          ::= ε

# === FUNCTION DEFINITION ===
function_def           ::= TOK_FUNCTION TOK_ID TOK_L_PAREN paramters_list TOK_R_PAREN TOK_ARROW return_type_definition body_def
paramters_list         ::= [ paramter { TOK_COMMA paramter } ] | ε
paramter               ::= TOK_ID TOK_COLON type_definition
body_def               ::= TOK_L_BRACE op_local_statement_list TOK_R_BRACE
return_stmt            ::= TOK_RETURN expr_value

# === TYPES ===
type_definition        ::= TOK_TYPE_INT | TOK_TYPE_STRING | TOK_TYPE_FLOAT | TOK_TYPE_BOOL | TOK_TYPE_LIST
return_type_definition ::= type_definition | TOK_TYPE_VOID

# === EXPRESSIONS & LITERALS ===
expr_value             ::= factor | sting_literal | bool_literal | TOK_ID
factor                 ::= TOK_NUMBER
sting_literal          ::= TOK_STRING
bool_literal           ::= TOK_TRUE | TOK_FALSE
